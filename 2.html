package com.ethiccode.chauhan;

import java.util.Scanner;

public class Assign2 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("\nEnter the number of forces: ");
        int n = scanner.nextInt();

        double[] weight = new double[n];
        double[] distance = new double[n];
        double beamLength, suptLeft, suptRight, totalWeight = 0, weightedMoment = 0;

        System.out.print("Enter the total length of the beam in cm: ");
        beamLength = scanner.nextDouble() / 100;
        System.out.print("\n");
        for (int i = 0; i < n; i++) {
            System.out.print("Enter force " + (i + 1) + " in Newtons: ");
            weight[i] = scanner.nextDouble() * 9.806;
        }
        System.out.print("\n");
        for (int i = 0; i < n; i++) {
            System.out.print("Enter the distance of force " + (i + 1) + " from the left end in cm: ");
            distance[i] = scanner.nextDouble() / 100;
        }
        System.out.print("\n");
        for (int i = 0; i < n; i++) {
            totalWeight += weight[i];
            weightedMoment += weight[i] * distance[i];
        }

        suptRight = weightedMoment / beamLength;
        suptLeft = totalWeight - suptRight;

        System.out.printf("\nThe support reaction on the right end is %.3f kg\n", suptRight / 10);
        System.out.printf("The support reaction on the left end is %.3f kg\n", suptLeft / 10);

        scanner.close();
    }
}
