package com.ethiccode.chauhan;
import java.util.Scanner;

public class Assgn1 {
    private static void calcErr(double[] forces, double[] angles) {
        double Fx = 0, Fy=0, resultant, angle;
        for (int i = 0; i < forces.length; i++) {
            double radian = Math.toRadians(angles[i]);
            Fx = forces[i] * Math.cos(radian);
            Fy = forces[i] * Math.sin(radian);
        }
        resultant = Math.sqrt(Fx * Fx + Fy * Fy);
        angle = Math.toDegrees(Math.atan2(Fy, Fx));

        System.out.println(String.format("\nThe Summation of All the Forces in X-direction is: %.3f",Fx));
        System.out.println(String.format("The Summation of All the Forces in Y-direction is: %.3f",Fy));
        System.out.println(String.format("The Resultant of the given Force System is: %.3f",resultant));
        System.out.println(String.format("The angle of Resultant with the abscissa is: %.3f",angle));

    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double[] forces = new double[5];
        double[] angles = new double[5];

        System.out.println("\nEnter 5 forces:");
        for (int i = 0; i < 5; i++) {
            System.out.print("F"+(i+1)+": ");
            forces[i] = sc.nextDouble();
        }

        System.out.println("\nEnter 5 angles, wrt x axis:");
        for (int i = 0; i < 5; i++) {
            System.out.print("A"+(i+1)+": ");
            angles[i] = sc.nextDouble();
        }

        calcErr(forces, angles);
        sc.close();
    }
}
